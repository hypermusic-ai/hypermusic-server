117 evmc::Result call(const evmc_message& msg) noexcept override
118 {
119 if (msg.kind == EVMC_CALL || msg.kind == EVMC_DELEGATECALL || msg.kind == EVMC_CALLCODE)
120 {
121 auto it = _accounts.find(to_key(msg.recipient));
122 if (it == _accounts.end())
123 return evmc::Result{EVMC_FAILURE};
124
125 auto& code = it->second.code;
126 if (code.empty())
127 return evmc::Result{EVMC_FAILURE};
128
129 return _vm->execute(*this, _revision, msg, code.data(), code.size());
130 }
131 else if (msg.kind == EVMC_CREATE || msg.kind == EVMC_CREATE2)
132 {
133 evmc::Result result = _vm->execute(*this, _revision, msg, msg.input_data, msg.input_size);
134 if (result.status_code != EVMC_SUCCESS)
135 return result;
136
137 std::vector<uint8_t> deployed_code(result.output_data, result.output_data + result.output_size);
138
139 evmc_address new_address;

141 {
142 new_address = derive_create_address(msg.sender, _create_nonce[msg.sender]++);
143 }
144 else // CREATE2
145 {
146 new_address = derive_create2_address(msg.sender, msg.create2_salt, deployed_code);
147 }
148
149 deploy_contract(new_address, std::move(deployed_code), msg.value); // implement this function to store the new contract
150 result.create_address = new_address;

152 }
