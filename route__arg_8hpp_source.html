294 std::optional<TupleT> parseRouteArgAs(const RouteArg& arg)
295 {
296 if(arg.getType() != RouteArgType::object)return std::nullopt;
297 if(std::tuple_size<TupleT>::value == 0) return std::nullopt;
298 if(arg.getChildren().size() != std::tuple_size<TupleT>::value) return std::nullopt;
299
300 std::string data = arg.getData();
301 if(data.empty())return std::nullopt;
302 if(data.front() != OBJECT_START_IDENTIFIER) return std::nullopt;
303 if(data.back() != OBJECT_END_IDENTIFIER) return std::nullopt;
304
305 // remove delimiters
306 data = data.substr(1, data.size() - 2);
307
308 // split arg into fields
309 std::vector<std::string> values_str;
310 std::size_t begin = 0;
311 std::size_t end = 0;
312
313 while ((end = data.find(OBJECT_FIELDS_DELIMETER, begin)) != std::string::npos) {
314 values_str.push_back(data.substr(begin, (end - begin)));
315 begin = end + 1;
316 }
317 // add the last value
318 values_str.push_back(data.substr(begin));
319
320 return TupleParser<TupleT, std::tuple_size<TupleT>::value>{}(arg.getChildren(), values_str);
321 }

323 template<IsSequenceContainer ContainerT>
324 std::optional<ContainerT> parseRouteArgAs(const RouteArg& arg)
325 {
326 using T = typename ContainerT::value_type;
327
328 if(arg.getType() != RouteArgType::array)return std::nullopt;
329 if(arg.getChildren().size() != 1)return std::nullopt;
330 if(arg.getChildren().at(0) == nullptr)return std::nullopt;
331
332 std::string data = arg.getData();
333 if(data.empty())return std::nullopt;
334 if(data.front() != ARRAY_START_IDENTIFIER) return std::nullopt;
335 if(data.back() != ARRAY_END_IDENTIFIER) return std::nullopt;
336
337 // remove delimiters
338 data = data.substr(1, data.size() - 2);
339
340 // split arg into values
341 constexpr static const char array_values_delimeter = ',';
342
343 std::vector<std::string> values_str;
344 std::size_t begin = 0;
345 std::size_t end = 0;
346
347 while ((end = data.find(array_values_delimeter, begin)) != std::string::npos) {
348 values_str.push_back(data.substr(begin, (end - begin)));
349 begin = end + 1;
350 }
351 // add the last value
352 values_str.push_back(data.substr(begin));
353
354 ContainerT values;
355 const auto & array_type = *arg.getChildren().at(0);
356
357 for (const auto & value_str : values_str)
358 {
359 RouteArg array_value = RouteArg(array_type, value_str);
360 const auto parsed_value = parseRouteArgAs<T>(array_value);
361 if(!parsed_value)return std::nullopt;
362 values.emplace_back(*parsed_value);
363 }
364
365 return values;
